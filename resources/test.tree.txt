axiom => ^{ BRANCH }$

BRANCH => { RY(-10,10) RZ(-30,30) BRANCH }

BRANCH => { RY(-10,10) RX(-50,-40) RZ(-20,20) BRANCH } { RY(-10,10) RX(40,50) RZ(-20,20) BRANCH }




^ ->
	rotateY(rnd(0,360));
	var tree = extrude({ width:5, ring:1, data:{ length:15 } }).setId(1)
	;

RGB(1,2,3) -> 
	tree.color(1,2,3)
	;

RX(-,+) -> 
	tree.rotateX(rnd(-,+))
	;

RY(-,+) -> 
	tree.rotateY(rnd(-,+))
	;

RZ(-,+) -> 
	tree.rotateZ(rnd(-,+))
	;

UP(A) ->
	tree.translate(0,A,0)
	;

{ ->
	tree = tree.extrude().mulWidth(0.6180339887);
	tree.data.length *= 0.618033;
	tree.translate(0, tree.data.length, 0)
	;

} ->
	tree = tree.parent
	;

$ -> ;
